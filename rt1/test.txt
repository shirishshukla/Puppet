  - name: Backup dirs if present 
    shell: |
        if [[  -d {{ item.key }} ]]; then 
            cp -rp {{ item.key }} {{ item.key }}.back
            rm -rf {{ item.key }}/*
        fi
        echo true
    with_dict: "{{ extra_hdd[0].lvs }}"
    ignore_errors: yes

    - name: Mount additional disks
      shell: |
          mount -a || true 
          if [[  -d {{ item.key }}.back ]]; then 
              rsync {{ item.key }}.back/ {{ item.key }}/
              rm -rf {{ item.key }}.back
          fi
          echo true
      with_dict: "{{ extra_hdd[0].lvs }}"
